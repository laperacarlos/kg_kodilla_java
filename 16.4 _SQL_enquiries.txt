select I.*, IL.NAME
from ISSUES I join ISSUELISTS IL on I.ISSUELIST_ID = IL.ID;

select I.*, U.FIRSTNAME as ASSIGNED_USER_FIRSTNAME, U.LASTNAME as ASSIGNED_USER_LASTNAME 
from ISSUES I join USERS U on U.ID = I.USER_ID_ASSIGNEDTO
order by I.ID;

select U.ID, U.FIRSTNAME, U.LASTNAME, count(*) as ASSIGNED_ISSUES
from ISSUES I join USERS U on U.ID = I.USER_ID_ASSIGNEDTO
group by U.ID;
